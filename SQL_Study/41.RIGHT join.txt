RIGHT Join :- 
     	    Returns all records from Right table, and matched records from left table 		 (-)     (-)         
    	         								        	( A(|=|)B|=|)	
		 										 (_)     (_)	
	Syntax :-
	SELECT column(s)        								  RIGHT JOIN
	FROM tableA
	RIGHT JOIN tableB
	ON tableA.col_name = tableB.col_name;
