LEFT Join :- 
     	    Returns all records from left table, and matched records from right table 		 (-)     (-)         
    	         								        	(|=|A(|=|)B )	
		 										 (_)     (_)	
	Syntax :-
	SELECT column(s)        								  LEFT JOIN
	FROM tableA
	LEFT JOIN tableB
	ON tableA.col_name = tableB.col_name;
